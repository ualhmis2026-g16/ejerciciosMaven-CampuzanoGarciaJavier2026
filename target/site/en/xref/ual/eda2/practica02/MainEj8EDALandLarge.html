7   	/**
8   	 * metodo main que se encarga de realizar la ejecucion de DijkstraEj8
9   	 * con el fichero graphEDAlandLarge
10  	 * @param Strin[] args
11  	 */
12      public static void main(String[] args) {
13          Graph graph = new Graph();
14          Map<String, Vertex> vertexMap = new HashMap<>();
15  
16          // Directorio de entrada
17          String directorioEntrada = System.getProperty("user.dir") + File.separator +
18                  "src" + File.separator +
19                  "main" + File.separator +
20                  "java" + File.separator +
21                  "ual" + File.separator +
22                  "eda2" + File.separator +
23                  "practica02" + File.separator;
24  
25          String filename = directorioEntrada + "graphEDAlandLarge.txt";
26  
27          // Leer archivo de grafo y construirlo
28          try (BufferedReader br = new BufferedReader(new FileReader(filename))) {
29              String line;
30              while ((line = br.readLine()) != null) {
31                  if (line.startsWith("#") || line.trim().isEmpty()) continue;
32  
33                  String[] parts = line.trim().split("\\s+");
34                  if (parts.length < 3) continue;
35  
36                  String fromId = parts[0];
37                  String toId = parts[1];
38                  double weight = Double.parseDouble(parts[2]);
39  
40                  Vertex from = vertexMap.computeIfAbsent(fromId, Vertex::new);
41                  Vertex to = vertexMap.computeIfAbsent(toId, Vertex::new);
42  
43                  graph.addEdge(from, to, weight);
44              }
45          } catch (IOException e) {
46              System.err.println("Error leyendo el archivo del grafo: " + e.getMessage());
47              return;
48          }
49  
50          // Solicitar al usuario un vértice de origen, destino y opcionalmente un intermedio
51          Scanner scanner = new Scanner(System.in);
52          System.out.print("Introduce el ID del vértice de origen: ");
53          String origenId = scanner.nextLine().trim();
54  
55          System.out.print("Introduce el ID del vértice de destino: ");
56          String destinoId = scanner.nextLine().trim();
57  
58          System.out.print("Introduce el ID del vértice intermedio (opcional, presiona Enter para omitir): ");
59          String intermedioId = scanner.nextLine().trim();
60  
61          // Verificar si el vértice de origen y destino existen
62          Vertex source = vertexMap.get(origenId);
63          Vertex destination = vertexMap.get(destinoId);
64          Vertex intermedio = intermedioId.isEmpty() ? null : vertexMap.get(intermedioId);
65  
66          if (source == null) {
67              System.err.println("El vértice de origen no existe en el grafo.");
68              return;
69          }
70          if (destination == null) {
71              System.err.println("El vértice de destino no existe en el grafo.");
72              return;
73          }
74          if (intermedioId != null && intermedio == null) {
75              System.err.println("El vértice intermedio no existe en el grafo.");
76              return;
77          }
78  
79          // Ejecutar el algoritmo de DijkstraEj8
80          TreeMap<String, Object> results;
81          double distanciaTotal = 0.0;
82          if (intermedio != null) {
83              System.out.println("\nCalculando camino mínimo desde " + source.getId() + " pasando por " + intermedio.getId() + " hasta " + destination.getId() + "...\n");
84              results = DijkstraEj8.DijkstraIntermedio(graph, source, intermedio, destination);
85  
86              @SuppressWarnings("unchecked")
87              TreeMap<Vertex, Double> distFuente = (TreeMap<Vertex, Double>) results.get("distFuente");
88              @SuppressWarnings("unchecked")
89              TreeMap<Vertex, Double> distIntermedio = (TreeMap<Vertex, Double>) results.get("distIntermedio");
90  
91              // Verificar que ambos tramos tienen un camino válido
92              if (distFuente.get(intermedio) == Double.POSITIVE_INFINITY || distIntermedio.get(destination) == Double.POSITIVE_INFINITY) {
93                  System.out.println("No existe un camino desde " + source.getId() + " hasta " + destination.getId() + " pasando por " + intermedio.getId());
94                  return;
95              }
96  
97              // Calcular la distancia total
98              distanciaTotal = distFuente.get(intermedio) + distIntermedio.get(destination);
99          } else {
100             System.out.println("\nCalculando camino mínimo desde " + source.getId() + " hasta " + destination.getId() + "...\n");
101             results = DijkstraEj8.Dijkstra(graph, source);
102 
103             @SuppressWarnings("unchecked")
104             TreeMap<Vertex, Double> dist = (TreeMap<Vertex, Double>) results.get("dist");
105 
106             // Obtener la distancia al destino
107             distanciaTotal = dist.get(destination);
108             if (distanciaTotal == Double.POSITIVE_INFINITY) {
109                 System.out.println("No existe un camino desde " + source.getId() + " hasta " + destination.getId());
110                 return;
111             }
112         }
113 
114         // Reconstruir el camino completo
115         List<Vertex> path = new ArrayList<>();
116         if (intermedio != null) {
117             @SuppressWarnings("unchecked")
118             TreeMap<Vertex, Vertex> predFuente = (TreeMap<Vertex, Vertex>) results.get("predFuente");
119             @SuppressWarnings("unchecked")
120             TreeMap<Vertex, Vertex> predIntermedio = (TreeMap<Vertex, Vertex>) results.get("predIntermedio");
121 
122             path.addAll(reconstructPath(predFuente, source, intermedio));
123             path.remove(path.size() - 1); // Evitar duplicar el nodo intermedio
124             path.addAll(reconstructPath(predIntermedio, intermedio, destination));
125         } else {
126             @SuppressWarnings("unchecked")
127             TreeMap<Vertex, Vertex> pred = (TreeMap<Vertex, Vertex>) results.get("pred");
128             path = reconstructPath(pred, source, destination);
129         }
130 
131         // Mostrar resultados
132         System.out.println("Camino mínimo encontrado:");
133         System.out.printf("Distancia total: %.2f\n", distanciaTotal);
134         System.out.print("Ruta: ");
135         for (int i = 0; i < path.size(); i++) {
136             System.out.print(path.get(i).getId());
137             if (i < path.size() - 1) {
138                 System.out.print(" -> ");
139             }
140         }
141         System.out.println();
142 
143         // Mostrar estadísticas
144         long tiempoEjecucion = (long) results.get("tiempo_ejecucion_ns");
145         int iteraciones = (int) results.get("iteraciones");
146         int actualizaciones = (int) results.get("actualizaciones");
147         int verticesProcesados = (int) results.get("vertices_procesados");
148         int aristasProcesadas = (int) results.get("aristas_procesadas");
149 
150         System.out.println("\nEstadísticas de ejecución:");
151         System.out.printf("Tiempo de ejecución: %d ns\n", tiempoEjecucion);
152         System.out.printf("Iteraciones: %d\n", iteraciones);
153         System.out.printf("Actualizaciones de distancia: %d\n", actualizaciones);
154         System.out.println("Vértices procesados: " + verticesProcesados);
155         System.out.println("Aristas procesadas: " + aristasProcesadas);
156 
157         scanner.close();
158     }

160     /**
161      * Reconstruye el camino desde el origen hasta el destino
162      * utilizando el mapa de predecesores.
163      *
164      * @param pred        Mapa de predecesores para cada vértice.
165      * @param source      Vértice de origen.
166      * @param destination Vértice de destino.
167      * @return Lista de vértices que forman el camino.
168      */
169     private static List<Vertex> reconstructPath(TreeMap<Vertex, Vertex> pred, Vertex source, Vertex destination) {
170         List<Vertex> path = new ArrayList<>();
171         Vertex current = destination;
172 
173         // Verificar si el destino es alcanzable
174         while (current != null && current != source) {
175             path.add(0, current);
176             current = pred.get(current);
177         }
178 
179         if (current == source) {
180             path.add(0, source);
181         }
182 
183         return path;
184     }
