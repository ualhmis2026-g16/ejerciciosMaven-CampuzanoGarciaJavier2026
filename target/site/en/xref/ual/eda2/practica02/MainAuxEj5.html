110     /**
111      * Carga las posiciones de los vértices desde un archivo
112      * para usar en la función heurística.
113      * Formato del archivo:
114      * - Líneas iniciales con número de vértices y aristas (ignoradas)
115      * - Lista de posiciones: id_vertice x y
116      * - Lista de aristas al final (ignoradas)
117      * @param String filePath
118      * @param Graph graph
119      * @return Map<Vertex, double[]>
120      */
121     private static Map<Vertex, double[]> loadPositions(String filePath, Graph graph) {
122         Map<Vertex, double[]> positions = new HashMap<>();
123 
124         try {
125             Scanner scanner = new Scanner(new File(filePath));
126             
127             // Ignorar las dos primeras líneas (número de vértices y aristas)
128             if (scanner.hasNextLine()) scanner.nextLine(); // Ignorar línea 1
129             if (scanner.hasNextLine()) scanner.nextLine(); // Ignorar línea 2
130 
131             // Leer posiciones de los vértices
132             while (scanner.hasNextLine()) {
133                 String line = scanner.nextLine().trim();
134 
135                 // Detener la lectura al encontrar la lista de aristas
136                 if (line.matches("^[A-Za-z]+\\s+[A-Za-z]+\\s+\\d+$")) {
137                     break; // Línea coincide con el formato de una arista
138                 }
139 
140                 // Dividir la línea en partes
141                 String[] parts = line.split("\\s+");
142                 if (parts.length != 3) {
143                     System.out.println("Línea inválida en el archivo de posiciones: " + line);
144                     continue; // Saltar líneas mal formateadas
145                 }
146 
147                 String id = parts[0];
148                 try {
149                     double x = Double.parseDouble(parts[1]);
150                     double y = Double.parseDouble(parts[2]);
151 
152                     // Buscar el vértice correspondiente en el grafo
153                     Vertex vertex = null;
154                     for (Vertex v : graph.getVertices()) {
155                         if (v.getId().equals(id)) {
156                             vertex = v;
157                             break;
158                         }
159                     }
160 
161                     if (vertex == null) {
162                         System.out.println("Vértice desconocido en el archivo de posiciones: " + id);
163                         continue;
164                     }
165 
166                     // Asignar las coordenadas al vértice
167                     positions.put(vertex, new double[]{x, y});
168                 } catch (NumberFormatException e) {
169                     System.out.println("Error al analizar coordenadas en la línea: " + line);
170                 }
171             }
172 
173             scanner.close();
174         } catch (IOException e) {
175             System.err.println("Error al cargar posiciones: " + e.getMessage());
176             generateRandomPositions(graph, positions);
177         }
178 
179         return positions;
180     }

182     /**
183      * Genera posiciones aleatorias para los vértices en caso de que el archivo de posiciones no exista o sea inválido.
184      * @param Graph graph
185      * @param Map<Vertex, double[]> positions
186      */
187     private static void generateRandomPositions(Graph graph, Map<Vertex, double[]> positions) {
188         Random random = new Random(42); // Semilla fija para reproducibilidad
189         for (Vertex v : graph.getVertices()) {
190             double x = random.nextDouble() * 1000;
191             double y = random.nextDouble() * 1000;
192             positions.put(v, new double[]{x, y});
193         }
194     }
