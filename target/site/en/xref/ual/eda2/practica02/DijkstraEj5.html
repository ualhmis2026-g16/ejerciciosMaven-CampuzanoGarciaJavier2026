11  	/**
12       * Clase interna para representar un nodo en el algoritmo A*
13       * Almacena información adicional necesaria para A*
14       */
15      private static class AStarNode implements Comparable<AStarNode> {
16          Vertex vertex;    // El vértice asociado a este nodo
17          double g;         // Coste real desde el origen hasta este nodo
18          double h;         // Valor heurístico (distancia estimada hasta el destino)
19          double f;         // Valor f = g + h
20          AStarNode parent; // Nodo padre para reconstruir la ruta
21          
22          public AStarNode(Vertex v) {
23              this.vertex = v;
24              this.g = Double.POSITIVE_INFINITY;
25              this.h = 0;
26              this.f = Double.POSITIVE_INFINITY;
27              this.parent = null;
28          }
29          
30          @Override
31          public int compareTo(AStarNode other) {
32              return Double.compare(this.f, other.f);
33          }
34          
35          @Override
36          public boolean equals(Object obj) {
37              if (this == obj) return true;
38              if (obj == null || getClass() != obj.getClass()) return false;
39              AStarNode other = (AStarNode) obj;
40              return vertex.equals(other.vertex);
41          }
42          
43          @Override
44          public int hashCode() {
45              return vertex.hashCode();
46          }
47      }

49      /**
50       * Algoritmo A* para encontrar la ruta mínima entre dos vértices
51       * @param G Grafo con los vértices y aristas
52       * @param source Vértice origen
53       * @param destination Vértice destino
54       * @param positions Mapa con las coordenadas (x,y) de cada vértice para la heurística
55       * @return Mapa con la ruta, distancia y estadísticas
56       */
57      public static TreeMap<String, Object> findPath(Graph G, Vertex source, Vertex destination, 
58                                                Map<Vertex, double[]> positions) {
59          // Variables para medir rendimiento
60          long startTime = System.nanoTime();
61          int nodesEvaluated = 0;
62          int nodesGenerated = 0;
63          
64          // Mapeo de vértices a nodos A*
65          TreeMap<Vertex, AStarNode> nodeMap = new TreeMap<>();
66          
67          // Inicializar listas abierta y cerrada
68          PriorityQueue<AStarNode> openList = new PriorityQueue<>(); // Nodos por evaluar
69          TreeSet<AStarNode> openSet = new TreeSet<>(); // Para búsqueda rápida en openList
70          TreeSet<AStarNode> closedList = new TreeSet<>(); // Nodos ya evaluados
71          
72          // Inicializar nodo origen
73          AStarNode sourceNode = new AStarNode(source);
74          sourceNode.g = 0;
75          sourceNode.h = heuristic(source, destination, positions);
76          sourceNode.f = sourceNode.g + sourceNode.h;
77          nodeMap.put(source, sourceNode);
78          
79          // Añadir nodo origen a la lista abierta
80          openList.add(sourceNode);
81          openSet.add(sourceNode);
82          nodesGenerated++;
83          
84          AStarNode current = null;
85          boolean pathFound = false;
86          
87          // Mientras la lista abierta no esté vacía
88          while (!openList.isEmpty()) {
89              // Obtener nodo con menor valor f
90              current = openList.poll();
91              openSet.remove(current);
92              nodesEvaluated++;
93              
94              // Comprobar si hemos llegado al destino
95              if (current.vertex.equals(destination)) {
96                  pathFound = true;
97                  break;
98              }
99              
100             // Mover nodo actual de lista abierta a lista cerrada
101             closedList.add(current);
102             
103             // Comprobar todos los nodos vecinos
104             for (Vertex neighborVertex : G.getVertices()) {
105                 if (G.adyacentes(current.vertex, neighborVertex)) {
106                     // Obtener o crear nodo vecino
107                     AStarNode neighbor = nodeMap.get(neighborVertex);
108                     if (neighbor == null) {
109                         neighbor = new AStarNode(neighborVertex);
110                         nodeMap.put(neighborVertex, neighbor);
111                         nodesGenerated++;
112                     }
113                     
114                     // Saltar nodos ya evaluados
115                     if (closedList.contains(neighbor)) {
116                         continue;
117                     }
118                     
119                     // Calcular g tentativo
120                     double tentativeG = current.g + G.weight(current.vertex, neighborVertex);
121                     
122                     boolean isNewNode = !openSet.contains(neighbor);
123                     
124                     if (isNewNode || tentativeG < neighbor.g) {
125                         // Este camino es mejor
126                         neighbor.parent = current;
127                         neighbor.g = tentativeG;
128                         neighbor.h = heuristic(neighborVertex, destination, positions);
129                         neighbor.f = neighbor.g + neighbor.h;
130                         
131                         if (isNewNode) {
132                             openList.add(neighbor);
133                             openSet.add(neighbor);
134                         } else {
135                             // Actualizar la posición en la cola de prioridad
136                             openList.remove(neighbor);
137                             openList.add(neighbor);
138                         }
139                     }
140                 }
141             }
142         }
143         
144         // Calcular estadísticas
145         long endTime = System.nanoTime();
146         long duration = endTime - startTime;
147         
148         // Reconstruir ruta si se encontró un camino
149         ArrayList<Vertex> path = new ArrayList<>();
150         double distance = Double.POSITIVE_INFINITY;
151         
152         if (pathFound) {
153             // Reconstruir camino
154             for (AStarNode node = current; node != null; node = node.parent) {
155                 path.add(0, node.vertex);
156             }
157             distance = current.g;
158         }
159         
160         // Preparar resultado
161         TreeMap<String, Object> result = new TreeMap<>();
162         result.put("path", path);
163         result.put("distance", distance);
164         result.put("pathFound", pathFound);
165         result.put("tiempo_ejecucion_ns", duration);
166         result.put("nodos_evaluados", nodesEvaluated);
167         result.put("nodos_generados", nodesGenerated);
168         
169         return result;
170     }

172     /**
173      * Función heurística - Distancia Euclídea
174      * @param from Vértice origen
175      * @param to Vértice destino
176      * @param positions Mapa con las coordenadas de los vértices
177      * @return Valor heurístico (distancia estimada)
178      */
179     private static double heuristic(Vertex from, Vertex to, Map<Vertex, double[]> positions) {
180         // Si no hay posiciones disponibles, devolver 0 (sin heurística)
181         if (positions == null || !positions.containsKey(from) || !positions.containsKey(to)) {
182             return 0;
183         }
184         
185         double[] posFrom = positions.get(from);
186         double[] posTo = positions.get(to);
187         
188         // Calcular distancia euclídea
189         double dx = posFrom[0] - posTo[0];
190         double dy = posFrom[1] - posTo[1];
191         
192         return Math.sqrt(dx*dx + dy*dy);
193     }
