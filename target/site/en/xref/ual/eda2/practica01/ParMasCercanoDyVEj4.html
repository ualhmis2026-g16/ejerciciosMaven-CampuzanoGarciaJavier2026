14  	 * Comparador de puntos según la coordenada X
15  	 */
16  	static Comparator<Poin> compareX = new Comparator<Poin>() {
17  		@Override
18  		public int compare(Poin p1, Poin p2) {
19  			return Double.compare(p1.x, p2.x);
20  		}
21  	};
22  	
23  	/*
24  	 * Comparador de puntos según la coordenada Y
25  	 */
26  	static Comparator<Poin> compareY = new Comparator<Poin>() {
27  		@Override
28  		public int compare(Poin p1, Poin p2) {
29  			return Double.compare(p1.y, p2.y);
30  		}
31  	};

33  	/**
34       * Calcula la distancia euclídea entre un par de puntos dado
35       * 
36       * @param p1 Primer punto
37       * @param p2 Segundo punto
38       * @return Distancia euclídea entre los dos puntos
39       */
40  	public static double distancia(Poin p1, Poin p2) {
41  		return Math.sqrt((p1.x - p2.x) * (p1.x - p2.x) +
42  						(p1.y - p2.y) * (p1.y - p2.y));
43  	}
44  	
45  	/**
46       * Método para encontrar la distancia entre los puntos más cercanos 
47       * de una lista haciendo uso del algoritmo de Divide y Vencerás, con una primera mejora:
48       * Utilizar una franja que cubra todo el espacio en el que haya puntos cuya coordenada
49       * x no diste del centro una distancia mayor a delta.
50       * 
51       * @param listaPuntos Los puntos a comparar
52       * @return Distancia mínima entre los puntos más cercanos de la lista
53       */
54  	public static double parMasCercanoDivideandConquer(ArrayList<Poin> listaPuntos) {
55  		Collections.sort(listaPuntos, compareX);
56  		return parMasCercanoDivideandConquer(listaPuntos, 0, listaPuntos.size()-1);
57  	}

60  	/**
61       * Método privado recursivo para encontrar el par más cercano, con una mejora (franja, sin ordenar).
62       * 
63       * @param puntos   Lista de puntos ordenados por coordenada X.
64       * @param comienzo Índice inicial del array.
65       * @param fin      Índice final del array.
66       * @return Distancia mínima entre los puntos más cercanos de la lista
67       */
68  	private static double parMasCercanoDivideandConquer(ArrayList<Poin> puntos, int comienzo, int fin) {
69  		
70  		//Caso base, fuerza bruta cuando el array de puntos es de tamaño 3 o inferior.
71  		
72  		if(fin-comienzo <=3) {
73  			double minDist = Double.MAX_VALUE;
74  			double aux;
75  			for(int i = comienzo; i<=fin; i++) {
76  				for(int j=i+1; j<=fin; j++) {
77  					aux = distancia(puntos.get(i), puntos.get(j));
78  					if(aux<minDist) {
79  						minDist = aux;
80  					}
81  				}
82  			}
83  			return minDist;
84  		} else {
85  			
86  			//Dividir
87  			
88  	        int mitad = comienzo + (fin - comienzo) / 2;
89  	        double izquierda = parMasCercanoDivideandConquer(puntos, comienzo, mitad);
90  	        double derecha = parMasCercanoDivideandConquer(puntos, mitad + 1, fin);
91  	        
92  	        //Combinar
93  	        
94  	        double delta = Math.min(izquierda, derecha);
95  	        double linea = puntos.get(mitad).x;
96  	        
97  	        ArrayList<Poin> franja = new ArrayList<>();
98  	        for (int i = comienzo; i <= fin; i++) {
99  	            if (Math.abs(puntos.get(i).x - linea) < delta)
100 	                franja.add(puntos.get(i));
101 	        }
102 
103 	        double aux;
104 	        for (int i = 0; i < franja.size(); i++) {
105 	            for (int j = i + 1; j < franja.size(); j++) {
106 	                aux = distancia(franja.get(i), franja.get(j));
107 	                if (aux < delta) {
108 	                    delta = aux;
109 	                }
110 	            }
111 	        }
112 	        return delta;
113 	    }
114 	}
