8   	// A structure to represent a Point in 2D plane
9   	public static class Point {
10  		public int x;
11  		public int y;
12  	 
13  		Point(int x, int y) {
14  			this.x = x;
15  			this.y = y;
16  		}
17  	}
18  
19  	public static double distance(Point p1, Point p2) {
20  		return Math.sqrt((p1.x - p2.x) * (p1.x - p2.x) +
21  						(p1.y - p2.y) * (p1.y - p2.y));
22  	  }
23  	
24  	static class PointXComparator implements Comparator<Point> {
25  		// Needed to sort array of points according to X coordinate
26  		@Override
27  		public int compare(Point pointA, Point pointB) {
28  			return Integer.compare(pointA.x, pointB.x);
29  		}
30  	}
31  
32  	static class PointYComparator implements Comparator<Point> {	 
33  		// Needed to sort array of points according to Y coordinate
34  		@Override
35  		public int compare(Point pointA, Point pointB) {
36  			return Integer.compare(pointA.y, pointB.y);
37  		} 
38  	}

40      private Point best1, best2;
41      private double bestDistance = Double.POSITIVE_INFINITY;
42      
43      public void ClosestPair(Point[] points) {
44          if (points == null) throw new IllegalArgumentException("constructor argument is null");
45          for (int i = 0; i < points.length; i++) {
46              if (points[i] == null) throw new IllegalArgumentException("array element " + i + " is null");
47          }
48  
49          int n = points.length;
50          if (n <= 1) return;
51  
52          // sort by x-coordinate (breaking ties by y-coordinate via stability)
53          Point[] pointsByX = new Point[n];
54          for (int i = 0; i < n; i++)
55              pointsByX[i] = points[i];
56          Arrays.sort(pointsByX, new PointYComparator());
57          Arrays.sort(pointsByX, new PointXComparator());
58          
59          // check for coincident points
60          for (int i = 0; i < n-1; i++) {
61              if (pointsByX[i].equals(pointsByX[i+1])) {
62                  bestDistance = 0.0;
63                  best1 = pointsByX[i];
64                  best2 = pointsByX[i+1];
65                  return;
66              }
67          }
68  
69          // sort by y-coordinate (but not yet sorted)
70          Point[] pointsByY = new Point[n];
71          for (int i = 0; i < n; i++)
72              pointsByY[i] = pointsByX[i];
73  
74          // auxiliary array
75          Point[] aux = new Point[n];
76  
77          closest(pointsByX, pointsByY, aux, 0, n-1);
78      }
79      
80      // find closest pair of points in pointsByX[lo..hi]
81      // precondition:  pointsByX[lo..hi] and pointsByY[lo..hi] are the same sequence of points
82      // precondition:  pointsByX[lo..hi] sorted by x-coordinate
83      // postcondition: pointsByY[lo..hi] sorted by y-coordinate
84      public double closest(Point[] pointsByX, Point[] pointsByY, Point[] aux, int lo, int hi) {
85          if (hi <= lo) return Double.POSITIVE_INFINITY;
86  
87          int mid = lo + (hi - lo) / 2;
88          Point median = pointsByX[mid];
89  
90          // compute closest pair with both endpoints in left subarray or both in right subarray
91          double delta1 = closest(pointsByX, pointsByY, aux, lo, mid);
92          double delta2 = closest(pointsByX, pointsByY, aux, mid+1, hi);
93          double delta = Math.min(delta1, delta2);
94  
95          // merge back so that pointsByY[lo..hi] are sorted by y-coordinate
96          merge(pointsByY, aux, lo, mid, hi);
97  
98          // aux[0..m-1] = sequence of points closer than delta, sorted by y-coordinate
99          int m = 0;
100         for (int i = lo; i <= hi; i++) {
101             if (Math.abs(pointsByY[i].x - median.x) < delta)
102                 aux[m++] = pointsByY[i];
103         }
104 
105         // compare each point to its neighbors with y-coordinate closer than delta
106         for (int i = 0; i < m; i++) {
107             // a geometric packing argument shows that this loop iterates at most 7 times
108             for (int j = i+1; (j < m) && (aux[j].y - aux[i].y < delta); j++) {
109                 double distance = distance(aux[i], aux[j]);
110                 if (distance < delta) {
111                     delta = distance;
112                     if (distance < bestDistance) {
113                         bestDistance = delta;
114                         best1 = aux[i];
115                         best2 = aux[j];
116                         // StdOut.println("better distance = " + delta + " from " + best1 + " to " + best2);
117                     }
118                 }
119             }
120         }
121         return delta;
122     }
123 
124     // stably merge a[lo .. mid] with a[mid+1 ..hi] using aux[lo .. hi]
125     // precondition: a[lo .. mid] and a[mid+1 .. hi] are sorted subarrays
126     private void merge(Point[] a, Point[] aux, int lo, int mid, int hi) {
127         // copy to aux[]
128         for (int k = lo; k <= hi; k++) {
129             aux[k] = a[k];
130         }
131 
132         // merge back to a[]
133         int i = lo, j = mid+1;
134         for (int k = lo; k <= hi; k++) {
135             if      (i > mid)              a[k] = aux[j++];
136             else if (j > hi)               a[k] = aux[i++];
137             else if (aux[j].y < aux[i].y)  a[k] = aux[j++];
138             else                           a[k] = aux[i++];
139         }
140     }
