3   import java.util.ArrayList;
4   import java.util.HashMap;
5   import java.util.Map;
6   
7   /**
8    * Clase que proporciona métodos para buscar caminos simples (Hamiltonianos)
9    * en un grafo, es decir, caminos que visitan cada vértice al menos una vez y
10   * regresan al vértice de origen.
11   */
12  public class CaminosSimples {
13  
14      /**
15       * Busca todos los caminos simples (Hamiltonianos) en el grafo que parten y
16       * regresan al vértice de origen especificado.
17       *
18       * @param grafo  El grafo en el cual buscar los caminos.
19       * @param origen El vértice de origen y destino para los caminos.
20       * @return Un mapa que asocia cada camino (lista de vértices) a su peso total.
21       */
22      public static Map<ArrayList<Vertex>, Double> buscarCaminosSimples(Graph grafo, Vertex origen) {
23          Map<ArrayList<Vertex>, Double> conjuntoCaminos = new HashMap<>();
24          ArrayList<Vertex> camino = new ArrayList<>();
25          ArrayList<Vertex> visitados = new ArrayList<>();
26  
27          caminosSimples(grafo, origen, camino, visitados, conjuntoCaminos, origen, 0.0);
28  
29          return conjuntoCaminos;
30      }
31  
32      /**
33       * Método recursivo para explorar caminos simples (Hamiltonianos) en el grafo.
34       *
35       * @param grafo         El grafo en el cual se busca el camino.
36       * @param actual        El vértice actual en la exploración.
37       * @param camino        Lista que representa el camino recorrido hasta el momento.
38       * @param visitados     Lista de vértices ya visitados para asegurar la simplicidad.
39       * @param conjuntoCaminos Mapa que almacena los caminos encontrados y su peso total.
40       * @param origen        El vértice de origen, al cual se debe regresar para cerrar el ciclo.
41       * @param pesoActual    El peso acumulado del camino actual.
42       */
43      private static void caminosSimples(Graph grafo, Vertex actual, ArrayList<Vertex> camino,
44                                         ArrayList<Vertex> visitados, Map<ArrayList<Vertex>, Double> conjuntoCaminos,
45                                         Vertex origen, double pesoActual) {
46          visitados.add(actual);
47          camino.add(actual);
48  
49          // Caso base: Si se han visitado todos los vértices y hay un arco que regresa al origen, se encontró un ciclo.
50          if (visitados.size() == grafo.getVertices().size() && grafo.adyacentes(actual, origen)) {
51              camino.add(origen); // Cerrar el ciclo agregando el origen
52              double pesoFinal = pesoActual + grafo.peso(actual, origen); // Sumar el peso del último tramo
53              conjuntoCaminos.put(new ArrayList<>(camino), pesoFinal);
54              camino.remove(camino.size() - 1); // Quitar la última adición para seguir explorando otros caminos
55          } else {
56              // Recorrer los vecinos del grafo que no han sido visitados
57              for (Vertex vecino : grafo.getVertices()) {
58                  if (!visitados.contains(vecino) && grafo.adyacentes(actual, vecino)) {
59                      double pesoArista = grafo.peso(actual, vecino);
60                      caminosSimples(grafo, vecino, camino, visitados, conjuntoCaminos, origen, pesoActual + pesoArista);
61                  }
62              }
63          }
64  
65          // Retroceder (backtracking)
66          camino.remove(camino.size() - 1);
67          visitados.remove(visitados.size() - 1);
68      }
69  }
