9   	// TSP ITERATIVE
10  		/**
11  		 * Resuelve el problema del viajero (TSP) utilizando un enfoque iterativo 
12  		 * con programación dinámica.
13  		 * 
14  		 * @param g El Grafo completo representado como una matriz de adyacencia.
15  		 * @param startVertex El vértice de inicio desde el cual se inicia la búsqueda.
16  		 * @return La solución óptima encontrada para el TSP.
17  		 */
18  		public static Solucion tspIterative(Grafo g, Vertex startVertex) {
19  			Solucion sol = null;
20  			
21  			//Contempla el caso de 1 solo vértice para devolver una lista con ese elemento.
22  			if (g.getVertices().size() == 1) {
23  			    Vertex v = g.getVertices().iterator().next();
24  			    List<Vertex> singlePath = new ArrayList<>();
25  			    singlePath.add(v);
26  			    singlePath.add(v);
27  			    return new Solucion(singlePath, 0.0);
28  			}
29  			
30  			// Inicializa el coste mínimo de la ruta
31  			double minTourCost = Double.POSITIVE_INFINITY;
32  			
33  			// Lista que almacenará el recorrido final
34  			List<Integer> tour = new ArrayList<>();
35  			
36  			// Obtiene el índice del nodo inicial en el Grafo
37  			int start = g.getVertexIndex(startVertex);
38  
39  			// Convierte el Grafo a una matriz de distancias
40  			double[][] distance = g.mapaAMatriz();
41  			
42  			// Número de nodos en el Grafo
43  			int N = distance.length;
44  			
45  			// Estado final, cuando todos los nodos han sido visitados
46  			final int END_STATE = (1 << N) - 1;
47  			
48  			// Tabla para memorización de subproblemas
49  			Double[][] memo = new Double[N][1 << N];
50  
51  			// Inicializa la tabla memo con las distancias directas desde el nodo inicial
52  			for (int end = 0; end < N; end++) {
53  				if (end == start) continue;
54  				memo[end][(1 << start) | (1 << end)] = distance[start][end];
55  			}
56  
57  			// Iteración sobre todos los subconjuntos de nodos de tamaño r, desde 3 hasta N
58  			for (int r = 3; r <= N; r++) {
59  				// Itera sobre todas las combinaciones de tamaño r de nodos
60  				for (int subset : combinations(r, N)) {
61  					// Asegúrate de que el nodo inicial esté en el subconjunto
62  					if (notIn(start, subset)) continue;
63  
64  					// Itera sobre todos los nodos posibles como siguiente nodo
65  					for (int next = 0; next < N; next++) {
66  						// Si el nodo es el nodo inicial o no está en el subconjunto, saltamos
67  						if (next == start || notIn(next, subset)) continue;
68  
69  						// Calcula el nuevo subconjunto sin el nodo 'next'
70  						// usa el operador XOR (^) para "remover" el bit correspondiente al nodo next dentro del subconjunto (subset).
71  						int subsetWithoutNext = subset ^ (1 << next);
72  						
73  						// Variable para almacenar el coste mínimo encontrado
74  						double minDist = Double.POSITIVE_INFINITY;
75  						
76  						// Busca el nodo anterior en el subconjunto que minimiza la distancia
77  						for (int end = 0; end < N; end++) {
78  							// Si el nodo 'end' es inválido, lo saltamos
79  							if (end == start || end == next || notIn(end, subset)) continue;
80  
81  							// Calcula la distancia entre el nodo 'end' y 'next'
82  							double newDistance = memo[end][subsetWithoutNext] + distance[end][next];
83  							// Actualiza el coste mínimo si se encuentra una mejor ruta
84  							if (newDistance < minDist) {
85  								minDist = newDistance;
86  							}
87  						}
88  
89  						// Guarda el coste mínimo en la tabla de memoización
90  						memo[next][subset] = minDist;
91  					}
92  				}
93  			}
94  
95  			// Conecta la ruta de vuelta al nodo inicial y minimiza el coste
96  			for (int i = 0; i < N; i++) {
97  				if (i == start) continue;
98  				double tourCost = memo[i][END_STATE] + distance[i][start];
99  				if (tourCost < minTourCost) {
100 					minTourCost = tourCost;
101 				}
102 			}
103 
104 			// Reconstrucción del recorrido mínimo desde la tabla de memoización
105 			int lastIndex = start;
106 			int state = END_STATE;
107 			tour.add(start);
108 
109 			// Se reconstruye el recorrido nodo por nodo
110 			for (int i = 1; i < N; i++) {
111 				int index = -1;
112 				// Busca el siguiente nodo que minimiza el coste
113 				for (int j = 0; j < N; j++) {
114 					if (j == start || notIn(j, state)) continue;
115 					if (index == -1) index = j;
116 					double prevDist = memo[index][state] + distance[index][lastIndex];
117 					double newDist  = memo[j][state] + distance[j][lastIndex];
118 					if (newDist < prevDist) {
119 						index = j;
120 					}
121 				}
122 
123 				// Agrega el siguiente nodo al recorrido
124 				tour.add(index);
125 				// Actualiza el estado al quitar el nodo de la lista de visitados
126 				state = state ^ (1 << index);
127 				// Actualiza el índice del último nodo visitado
128 				lastIndex = index;
129 			}
130 
131 			// Agrega el nodo inicial al final del recorrido
132 			tour.add(start);
133 
134 			// Invierte el recorrido para obtenerlo en el orden correcto (de inicio a fin)
135 			//Collections.reverse(tour);
136 			
137 			// Convierte los índices de los nodos a vértices del Grafo
138 			List<Vertex> vertexes = g.obtenerListaVertices(tour.toArray(new Integer[tour.size()]));
139 			
140 			double tourCost = 0;
141 			for(int i=0; i<vertexes.size()-1; i++) {
142 				tourCost+= g.peso(vertexes.get(i), vertexes.get(i+1));
143 			}
144 			// Crea la solución con el recorrido y el coste mínimo
145 			sol = new Solucion(vertexes, tourCost);
146 			return sol;
147 		}

150 		/**
151 		 * Verifica si un elemento no está en un subconjunto representado por un entero.
152 		 * 
153 		 * @param elem El elemento a verificar.
154 		 * @param subset El subconjunto representado como un entero.
155 		 * @return true si el elemento no está en el subconjunto, false en caso contrario.
156 		 */
157 		private static boolean notIn(int elem, int subset) {
158 			// Verifica si el elemento 'elem' no está en el subconjunto (usando bits)
159 			return ((1 << elem) & subset) == 0;
160 		}

162 		/**
163 		 * Genera todas las combinaciones de tamaño r de un conjunto de n elementos.
164 		 * 
165 		 * @param r El tamaño de las combinaciones a generar.
166 		 * @param n El número total de elementos en el conjunto.
167 		 * @return Una lista de enteros representando las combinaciones generadas.
168 		 */
169 		private static List<Integer> combinations(int r, int n) {
170 			List<Integer> subsets = new ArrayList<>();
171 			combinations(0, 0, r, n, subsets);
172 			return subsets;
173 		}
174 
175 		/**
176 		 * Genera combinaciones de tamaño r a partir de un conjunto de n elementos.
177 		 * 
178 		 * @param set El conjunto actual representado como un entero.
179 		 * @param at El índice actual en el conjunto.
180 		 * @param r El tamaño de las combinaciones a generar.
181 		 * @param n El número total de elementos en el conjunto.
182 		 * @param subsets La lista donde se almacenan las combinaciones generadas.
183 		 */
184 		private static void combinations(int set, int at, int r, int n, List<Integer> subsets) {
185 			// Verifica si hay suficientes elementos restantes para elegir
186 			int elementsLeftToPick = n - at;
187 			if (elementsLeftToPick < r) return;
188 
189 			// Si se han seleccionado 'r' elementos, guarda el subconjunto
190 			if (r == 0) {
191 				subsets.add(set);
192 			} else {
193 				for (int i = at; i < n; i++) {
194 					// Incluye el elemento 'i' en el subconjunto
195 					set |= 1 << i;
196 					// Se llama recursivamente para incluir más elementos
197 					combinations(set, i + 1, r - 1, n, subsets);
198 					// Se deshace la inclusión para explorar el caso sin él
199 					set &= ~(1 << i);
200 				}
201 			}
202 		}
