50       * Verifica que se calculen correctamente las distancias mínimas desde el nodo Madrid.

52      @Test
53      void testDijkstraSimpleGraph() {
54          Graph graph = new Graph();
55          Vertex madrid = new Vertex("Madrid");
56          Vertex barcelona = new Vertex("Barcelona");
57          Vertex sevilla = new Vertex("Sevilla");
58          Vertex valencia = new Vertex("Valencia");
59  
60          graph.addEdge(madrid, barcelona, 5);
61          graph.addEdge(madrid, sevilla, 10);
62          graph.addEdge(barcelona, valencia, 3);
63          graph.addEdge(sevilla, valencia, 7);
64  
65          Vertex source = madrid;
66          Map<String, Object> result = DijkstraEj1.Dijkstra(graph, source);
67  
68          @SuppressWarnings("unchecked")
69          TreeMap<Vertex, Double> dist = (TreeMap<Vertex, Double>) result.get("dist");
70  
71          // Distancia a sí mismo debería ser null (o 0 según implementación)
72          assertEquals(null, dist.get(madrid));
73          assertEquals(5.0, dist.get(barcelona));
74          assertEquals(10.0, dist.get(sevilla));
75          assertEquals(8.0, dist.get(valencia)); // Camino más corto: Madrid -> Barcelona -> Valencia
76      }

108      */
109     @Test
110     void testDijkstraWithCycles() {
111         Graph graph = new Graph();
112         Vertex a = new Vertex("A");
113         Vertex b = new Vertex("B");
114         Vertex c = new Vertex("C");
115         Vertex d = new Vertex("D");
116 
117         graph.addEdge(a, b, 2);
118         graph.addEdge(b, c, 3);
119         graph.addEdge(c, a, 4); // Ciclo: A -> B -> C -> A
120         graph.addEdge(a, d, 1);
121         graph.addEdge(d, c, 1);
122 
123         Vertex source = a;
124         Map<String, Object> result = DijkstraEj1.Dijkstra(graph, source);
125 
126         @SuppressWarnings("unchecked")
127         TreeMap<Vertex, Double> dist = (TreeMap<Vertex, Double>) result.get("dist");
128 
129         assertEquals(null, dist.get(a)); // Nodo fuente
130         assertEquals(2.0, dist.get(b));
131         assertEquals(2.0, dist.get(c)); // Camino óptimo: A -> D -> C
132         assertEquals(1.0, dist.get(d));
133     }
