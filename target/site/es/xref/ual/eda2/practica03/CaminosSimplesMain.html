19       * Reads a graph from a text file.
20       * 
21       * @param filePath The path to the text file
22       * @return A Graph object constructed from the file data
23       * @throws IOException If an I/O error occurs
24       */
25  	  public static Graph readGraphFromFile(String filePath) throws IOException {
26  	        Graph graph = new Graph();
27  	        Map<String, Vertex> verticesMap = new HashMap<>();
28  	        
29  	        try (BufferedReader reader = new BufferedReader(new FileReader(filePath))) {
30  	            String line;
31  	            while ((line = reader.readLine()) != null) {
32  	                line = line.trim();
33  	                if (line.isEmpty()) continue;
34  	                
35  	                String[] parts = line.split("\\s+");
36  	                if (parts.length == 3) {
37  	                    String sourceId = parts[0];
38  	                    String destId = parts[1];
39  	                    double distance = Double.parseDouble(parts[2]);
40  	                    
41  	                    // Get or create vertices
42  	                    Vertex source = getOrCreateVertex(verticesMap, graph, sourceId);
43  	                    Vertex dest = getOrCreateVertex(verticesMap, graph, destId);
44  	                    
45  	                    // Add the edge to the graph (assuming undirected graph)
46  	                    graph.addEdge(source, dest, distance);
47  	                    graph.addEdge(dest, source, distance); // Remove this line if it's a directed graph
48  	                }
49  	            }
50  	        }
51  	        
52  	        return graph;
53  	    }
54  	    
55  	    /**
56  	     * Helper method to get an existing vertex or create a new one if needed
57  	     * @param Map<String, Vertex> verticesMap
58  	     * @param Graph graph
59  	     * @param String id
60  	     * @return Vertex
61  	     */
62  	    private static Vertex getOrCreateVertex(Map<String, Vertex> verticesMap, Graph graph, String id) {
63  	        Vertex vertex = verticesMap.get(id);
64  	        if (vertex == null) {
65  	            vertex = new Vertex(id);
66  	            verticesMap.put(id, vertex);
67  	            graph.addVertex(vertex);
68  	        }
69  	        return vertex;
70  	    }
