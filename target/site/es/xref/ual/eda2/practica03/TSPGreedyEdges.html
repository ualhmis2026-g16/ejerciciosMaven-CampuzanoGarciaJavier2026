1   package ual.eda2.practica03;
2   
3   import java.util.ArrayList;
4   import java.util.HashMap;
5   import java.util.HashSet;
6   import java.util.List;
7   import java.util.Map;
8   import java.util.Set;
9   
10  public class TSPGreedyEdges {
11  
12  	// TSP GREEDY EDGES
13  	/**
14  	 * Resuelve el problema del viajero (TSP) utilizando un enfoque voraz basado 
15  	 * en la selección de aristas.
16  	 * 
17  	 * @param g El grafo completo representado como una lista de adyacencia.
18  	 * @param startVertex El vértice de inicio desde el cual se inicia la búsqueda.
19  	 * @return La solución encontrada para el TSP utilizando el algoritmo voraz.
20  	 */
21  	public static Solucion tspGreedyEdges(Grafo g, Vertex startVertex) {
22  	    // Paso 1: Verifica si el vértice inicial es válido y está en el grafo
23  	    if (startVertex == null || !g.getVertices().contains(startVertex)) return null;
24  
25  	    // Paso 2: Obtener y ordenar todas las aristas del grafo
26  	    List<Edge> edges = new ArrayList<>();
27  	    for (Vertex v : g.getVertices()) {
28  	        for (Vertex u : g.getVecinos(v)) {
29  	            if (v.compareTo(u) < 0) { // Evitar duplicar aristas
30  	                edges.add(new Edge(v, u, g.peso(v, u)));
31  	            }
32  	        }
33  	    }
34  
35  	    // Ordenar las aristas por peso
36  	    edges.sort(null);
37  
38  	    // Paso 3: Inicializar estructuras para los grados y componentes conectados
39  	    Map<Vertex, Integer> degree = new HashMap<>();
40  	    Map<Vertex, Integer> component = new HashMap<>();
41  	    int compId = 0;
42  	    for (Vertex v : g.getVertices()) {
43  	        component.put(v, compId++);
44  	        degree.put(v, 0);
45  	    }
46  
47  	    Map<Vertex, List<Vertex>> selectedEdges = new HashMap<>();
48  	    int selectedEdgesCount = 0;
49  
50  	    // Paso 4: Seleccionar las aristas válidas para el ciclo
51  	    for (Edge e : edges) {
52  	    	Vertex u = e.getSource();
53  	    	Vertex v = e.getDestination();
54  
55  	        // Evitar aristas que generen vértices con más de 2 conexiones
56  	        if (degree.get(u) >= 2 || degree.get(v) >= 2) continue;
57  
58  	        boolean isLastEdge = selectedEdgesCount == g.size() - 1;
59  	        
60  	        // Evitar formar ciclos prematuros
61  	        if (component.get(u).equals(component.get(v)) && !isLastEdge) continue;
62  
63  	        // Añadir la arista al camino
64  	        // Así queda mas PRO .....pero mejor que no: selectedEdges.computeIfAbsent(u, _ -> new ArrayList<>()).add(v);
65  	        List<Vertex> list = selectedEdges.get(u);
66  	        if (list == null) selectedEdges.put(u, list = new ArrayList<>());
67  	        list.add(v);
68  	        //Igual que esta: selectedEdges.computeIfAbsent(v, _ -> new ArrayList<>()).add(u);
69  	        list = selectedEdges.get(v);
70  	        if (list == null) selectedEdges.put(v, list = new ArrayList<>());
71  	        list.add(u);
72  
73  	        selectedEdgesCount++;
74  	        degree.put(u, degree.get(u) + 1);
75  	        degree.put(v, degree.get(v) + 1);
76  
77  	        // Unir los componentes de los vértices
78  	        int oldComp = component.get(v);
79  	        int newComp = component.get(u);
80  	        for (Vertex w : component.keySet()) {
81  	            if (component.get(w) == oldComp) {
82  	                component.put(w, newComp);
83  	            }
84  	        }
85  
86  	        // Si es la última arista, terminamos
87  	        if (isLastEdge) break;
88  	    }
89  
90  	    // Verifica si se encontraron suficientes aristas para formar el ciclo
91  	    if (selectedEdgesCount != g.size()) throw new RuntimeException("No existe solución válida");
92  
93  	    // Paso 6: Recorrer el camino desde el vértice inicial
94  	    List<Vertex> path = new ArrayList<>();
95  	    Set<Vertex> visited = new HashSet<>();
96  	    double totalWeight = 0.0;
97  	    Vertex current = startVertex;
98  	    path.add(current);
99  	    visited.add(current);
100 
101 	    // Mientras el camino no esté completo, seguimos agregando vértices
102 	    while (path.size() < g.size()) {
103 	        for (Vertex neighbor : selectedEdges.get(current)) {
104 	            if (!visited.contains(neighbor)) {
105 	                path.add(neighbor);
106 	                totalWeight += g.peso(current, neighbor);
107 	                visited.add(neighbor);
108 	                current = neighbor;
109 	                break;
110 	            }
111 	        }
112 	    }
113 
114 	    // Cierra el ciclo agregando el vértice inicial al final del camino
115 	    path.add(startVertex);
116 	    totalWeight += g.peso(current, startVertex);
117 
118 	    // Devuelve la solución con el recorrido y el peso total
119 	    return new Solucion(path, totalWeight);
120 	}
121 }
