68      /**
69       * Lee un grafo desde un archivo de texto.
70       * 
71       * @param filePath Ruta del archivo de texto
72       * @return Un objeto Graph construido a partir de los datos del archivo
73       * @throws IOException Si ocurre un error de entrada/salida
74       */
75      public static Grafo readGraphFromFile(String filePath) throws IOException {
76          Grafo graph = new Grafo();
77          Map<String, Vertex> verticesMap = new HashMap<>();
78  
79          try (BufferedReader reader = new BufferedReader(new FileReader(filePath))) {
80              String line;
81              while ((line = reader.readLine()) != null) {
82                  line = line.trim();
83                  if (line.isEmpty()) continue;
84  
85                  String[] parts = line.split("\\s+");
86                  if (parts.length == 3) {
87                      String sourceId = parts[0];
88                      String destId = parts[1];
89                      double distance = Double.parseDouble(parts[2]);
90  
91                      // Obtener o crear vértices
92                      Vertex source = getOrCreateVertex(verticesMap, graph, sourceId);
93                      Vertex dest = getOrCreateVertex(verticesMap, graph, destId);
94  
95                      // Añadir las aristas al grafo (suponiendo grafo no dirigido)
96                      graph.addEdge(source, dest, distance);
97                      graph.addEdge(dest, source, distance); // Elimina esta línea si el grafo es dirigido
98                  }
99              }
100         }
101 
102         return graph;
103     }

105     /**
106      * Método auxiliar para obtener un vértice existente o crear uno nuevo si es necesario.
107      * 
108      * @param verticesMap Mapa de identificadores de vértices a objetos Vertex
109      * @param graph Grafo donde se añaden los vértices
110      * @param id Identificador del vértice
111      * @return Un objeto Vertex correspondiente al identificador
112      */
113     private static Vertex getOrCreateVertex(Map<String, Vertex> verticesMap, Grafo graph, String id) {
114         Vertex vertex = verticesMap.get(id);
115         if (vertex == null) {
116             vertex = new Vertex(id);
117             verticesMap.put(id, vertex);
118             graph.addVertex(vertex);
119         }
120         return vertex;
121     }
