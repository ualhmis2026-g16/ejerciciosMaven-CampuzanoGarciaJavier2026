12  /**
13   * La clase CaminoMasCorto proporciona métodos para encontrar el camino 
14   * más corto (en términos de peso total) en un grafo. Utiliza 
15   * la clase CaminosSimples para obtener todos los caminos simples (circuitos) 
16   * a partir de un vértice de origen y luego selecciona aquel de menor peso.

20      /**
21       * Obtiene el camino simple de menor peso a partir del vértice de origen en el grafo dado.
22       * Se recorre el conjunto de caminos simples obtenidos y se selecciona el camino con 
23       * la menor suma de pesos.
24       * 
25       * @param grafo  Grafo en el cual se busca el camino.
26       * @param source Vértice de origen desde el que se inician los caminos.
27       * @return El ArrayList de vértices que representa el camino más corto.
28       */
29      public static ArrayList<Vertex> shortestPath(Graph grafo, Vertex source) {
30          ArrayList<Vertex> caminoElegido = new ArrayList<>();
31          double peso = Double.POSITIVE_INFINITY;
32          Map<ArrayList<Vertex>, Double> conjuntoCaminos = CaminosSimples.buscarCaminosSimples(grafo, source);
33          
34          // Iterar sobre cada camino simple y su peso para seleccionar el de menor peso
35          for (Entry<ArrayList<Vertex>, Double> entry : conjuntoCaminos.entrySet()) {
36              if (entry.getValue() < peso) {
37                  caminoElegido = entry.getKey();
38                  peso = entry.getValue();
39              }
40          }
41          
42          System.out.println("El camino elegido es: " + caminoElegido.toString());
43          System.out.println();
44          System.out.println("El peso del camino elegido es: " + peso);
45          return caminoElegido;
46      }

76      /**
77       * Lee un grafo desde un archivo de texto.
78       * El archivo debe tener una línea por cada arista con tres elementos separados por espacios:
79       * el identificador del vértice origen, el identificador del vértice destino y el peso (distancia) de la arista.
80       * 
81       * @param filePath Ruta del archivo de texto.
82       * @return Un objeto Graph construido a partir de los datos del archivo.
83       * @throws IOException Si ocurre un error de entrada/salida.
84       */
85      public static Graph readGraphFromFile(String filePath) throws IOException {
86          Graph graph = new Graph();
87          Map<String, Vertex> verticesMap = new HashMap<>();
88  
89          try (BufferedReader reader = new BufferedReader(new FileReader(filePath))) {
90              String line;
91              while ((line = reader.readLine()) != null) {
92                  line = line.trim();
93                  if (line.isEmpty()) continue;
94  
95                  String[] parts = line.split("\\s+");
96                  if (parts.length == 3) {
97                      String sourceId = parts[0];
98                      String destId = parts[1];
99                      double distance = Double.parseDouble(parts[2]);
100 
101                     // Obtener o crear vértices
102                     Vertex source = getOrCreateVertex(verticesMap, graph, sourceId);
103                     Vertex dest = getOrCreateVertex(verticesMap, graph, destId);
104 
105                     // Añadir la arista al grafo (suponiendo grafo no dirigido)
106                     graph.addEdge(source, dest, distance);
107                     graph.addEdge(dest, source, distance); // Eliminar esta línea si el grafo es dirigido
108                 }
109             }
110         }
111 
112         return graph;
113     }

115     /**
116      * Método auxiliar que obtiene un vértice existente o crea uno nuevo si es necesario.
117      * 
118      * @param verticesMap Mapa de identificadores de vértices a objetos Vertex.
119      * @param graph       Grafo en el cual se añaden los vértices.
120      * @param id          Identificador del vértice.
121      * @return Un objeto Vertex correspondiente al identificador dado.
122      */
123     private static Vertex getOrCreateVertex(Map<String, Vertex> verticesMap, Graph graph, String id) {
124         Vertex vertex = verticesMap.get(id);
125         if (vertex == null) {
126             vertex = new Vertex(id);
127             verticesMap.put(id, vertex);
128             graph.addVertex(vertex);
129         }
130         return vertex;
131     }
