3   import java.util.PriorityQueue;
4   import java.util.TreeMap;
5   
6   public class DijkstraEj7 {
7   
8   	/**
9        * Clase interna para representar un elemento en la cola de prioridad
10       * Cada elemento contiene el vértice destino (to), la distancia y el vértice origen (from)
11       */
12      private static class VertexTriple implements Comparable<VertexTriple> {
13          Vertex to;       // Vértice destino
14          double distance; // Distancia acumulada hasta este vértice
15          Vertex from;     // Vértice origen (predecesor)
16          
17          public VertexTriple(Vertex to, double distance, Vertex from) {
18              this.to = to;
19              this.distance = distance;
20              this.from = from;
21          }
22          
23          @Override
24          public int compareTo(VertexTriple other) {
25              return Double.compare(this.distance, other.distance);
26          }
27      }
28      
29      /**
30       * Versión del algoritmo de Dijkstra usando Best-First Search
31       * No utiliza una estructura para almacenar vértices visitados
32       * @param G Grafo con los vértices y aristas
33       * @param fuente Vértice inicial
34       * @return Mapa con distancias y predecesores
35       */
36      public static TreeMap<String, Object> Dijkstra(Graph G, Vertex fuente) {
37          // Variables para medir rendimiento
38          long startTime = System.nanoTime();
39          int iteraciones = 0;
40          int actualizaciones = 0;
41          int verticesProcesados = 0;
42          int aristasProcesadas = 0;
43          
44          // Estructuras de datos
45          TreeMap<Vertex, Double> dist = new TreeMap<>();
46          TreeMap<Vertex, Vertex> pred = new TreeMap<>();
47          
48          // Cola de prioridad organizada por distancia mínima con tripletas <to, distance, from>
49          PriorityQueue<VertexTriple> pq = new PriorityQueue<>();
50          
51          // Inicializar distancias
52          for (Vertex v : G.getVertices()) {
53              if (v.equals(fuente)) {
54                  dist.put(v, Double.POSITIVE_INFINITY); // Distancia a la fuente es infinito
55              } else {
56                  dist.put(v, Double.NEGATIVE_INFINITY);
57              }
58              pred.put(v, null);
59          }
60          
61          // Añadir el vértice fuente a la cola
62          pq.add(new VertexTriple(fuente, Double.POSITIVE_INFINITY, null));
63          
64          // Mientras la cola de prioridad no esté vacía
65          while (!pq.isEmpty()) {
66              iteraciones++;
67              
68              // Extraer la tripleta con menor distancia
69              VertexTriple triple = pq.poll();
70              Vertex to = triple.to;
71              double ancho = triple.distance;
72              Vertex from = triple.from;
73              
74              // Si la distancia en la cola es mayor que la registrada, ignorar este camino
75              if (ancho < dist.get(to)) {
76                  continue;
77              }
78              
79              if(from!=null) {
80              	pred.put(to, from);
81              }
82              
83              verticesProcesados++;
84              
85              // Para cada vértice adyacente
86              for (Vertex neighbor : G.getVertices()) {
87                  if (G.adyacentes(to, neighbor)) {
88                      aristasProcesadas++;
89                      
90                      // Calcular nueva distancia
91                      double weight = G.weight(to, neighbor);
92                      double nuevaCapacidad = Math.min(dist.get(to), weight);
93                      
94                      // Verificar si hay un camino más corto
95                      if (nuevaCapacidad > dist.get(neighbor)) {
96                          // Actualizar distancia
97                          dist.put(neighbor, nuevaCapacidad);
98                          
99                          // Añadir a la cola con la nueva distancia y el vértice origen
100                         pq.add(new VertexTriple(neighbor, nuevaCapacidad, to));
101                         
102                         actualizaciones++;
103                     }
104                 }
105             }
106         }
107         
108         // Calcular estadísticas
109         long endTime = System.nanoTime();
110         long duration = endTime - startTime;
111         
112         // Prepara resultado
113         TreeMap<String, Object> result = new TreeMap<>();
114         result.put("dist", dist);
115         result.put("pred", pred);
116         result.put("tiempo_ejecucion_ns", duration);
117         result.put("iteraciones", iteraciones);
118         result.put("actualizaciones", actualizaciones);
119         result.put("vertices_procesados", verticesProcesados);
120         result.put("aristas_procesadas", aristasProcesadas);
121         
122         return result;
123     }
