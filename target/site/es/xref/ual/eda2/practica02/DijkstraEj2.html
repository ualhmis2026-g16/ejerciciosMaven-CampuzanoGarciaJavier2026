3   import java.util.TreeMap;
4   import java.util.TreeSet;
5   
6   public class DijkstraEj2 {
7   	/**
8        * Versión Naive (O(V²)) del algoritmo de Dijkstra
9        * @param G Grafo con los vértices y aristas
10       * @param fuente Vértice inicial
11       * @return Mapa con distancias y predecesores
12       */
13      public static TreeMap<String, Object> Dijkstra(Graph G, Vertex fuente) {
14          // Variables para medir rendimiento
15          long startTime = System.nanoTime();
16          int iteraciones = 0;
17          int actualizaciones = 0;
18          int verticesProcesados = 0;
19          int aristasProcesadas = 0;
20          
21          // Estructuras de datos
22          TreeMap<Vertex, Double> dist = new TreeMap<>();
23          TreeMap<Vertex, Vertex> pred = new TreeMap<>();
24          TreeSet<Vertex> visitados = new TreeSet<>();
25          
26          // Inicializar distancias
27          for (Vertex v : G.getVertices()) {
28              if (v.equals(fuente)) {
29                  dist.put(v, 0.0); // Distancia a la fuente es 0
30              } else if (G.adyacentes(fuente, v)) {
31                  dist.put(v, G.peso(fuente, v));
32                  pred.put(v, fuente);
33              } else {
34                  dist.put(v, Double.POSITIVE_INFINITY);
35                  pred.put(v, null);
36              }
37          }
38          
39          // Marcar la fuente como visitada
40          visitados.add(fuente);
41          
42          // Mientras no se hayan visitado todos los vértices
43          while (visitados.size() < G.getVertices().size()) {
44              iteraciones++;
45              
46              // Encontrar el vértice no visitado con menor distancia (from)
47              Vertex from = null;
48              double minDist = Double.POSITIVE_INFINITY;
49              
50              for (Vertex v : G.getVertices()) {
51                  if (!visitados.contains(v) && dist.get(v) < minDist) {
52                      minDist = dist.get(v);
53                      from = v;
54                  }
55              }
56              
57              // Si no hay camino a ningún vértice más, terminamos
58              if (from == null) {
59                  break;
60              }
61              
62              // Marcar 'from' como visitado
63              visitados.add(from);
64              verticesProcesados++;
65              
66              // Para cada vértice adyacente no visitado (to)
67              for (Vertex to : G.getVertices()) {
68                  if (!visitados.contains(to) && G.adyacentes(from, to)) {
69                  	aristasProcesadas++;
70                      // Verificar si hay un camino más corto
71                      if (dist.get(from) + G.weight(from, to) < dist.get(to)) {
72                          // Actualizar distancia y predecesor
73                          dist.put(to, dist.get(from) + G.weight(from, to));
74                          pred.put(to, from);
75                          actualizaciones++;
76                      }
77                  }
78              }
79          }
80          
81          // Calcular estadísticas
82          long endTime = System.nanoTime();
83          long duration = endTime - startTime;
84          
85          // Prepara resultado
86          TreeMap<String, Object> result = new TreeMap<>();
87          result.put("dist", dist);
88          result.put("pred", pred);
89          result.put("tiempo_ejecucion_ns", duration);
90          result.put("iteraciones", iteraciones);
91          result.put("actualizaciones", actualizaciones);
92          result.put("aristas_procesadas", aristasProcesadas);
93          result.put("vertices_procesados", verticesProcesados);
94          
95          return result;
96      }
