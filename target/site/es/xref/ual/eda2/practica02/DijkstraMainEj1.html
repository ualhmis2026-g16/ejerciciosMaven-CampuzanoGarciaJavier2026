17      /**
18       * Reads a graph from a text file.
19       * 
20       * @param filePath The path to the text file
21       * @return A Graph object constructed from the file data
22       * @throws IOException If an I/O error occurs
23       */
24  	  public static Graph readGraphFromFile(String filePath) throws IOException {
25  	        Graph graph = new Graph();
26  	        Map<String, Vertex> verticesMap = new HashMap<>();
27  	        
28  	        try (BufferedReader reader = new BufferedReader(new FileReader(filePath))) {
29  	            String line;
30  	            while ((line = reader.readLine()) != null) {
31  	                line = line.trim();
32  	                if (line.isEmpty()) continue;
33  	                
34  	                String[] parts = line.split("\\s+");
35  	                if (parts.length == 3) {
36  	                    String sourceId = parts[0];
37  	                    String destId = parts[1];
38  	                    double distance = Double.parseDouble(parts[2]);
39  	                    
40  	                    // Get or create vertices
41  	                    Vertex source = getOrCreateVertex(verticesMap, graph, sourceId);
42  	                    Vertex dest = getOrCreateVertex(verticesMap, graph, destId);
43  	                    
44  	                    // Add the edge to the graph (assuming undirected graph)
45  	                    graph.addEdge(source, dest, distance);
46  	                    graph.addEdge(dest, source, distance); // Remove this line if it's a directed graph
47  	                }
48  	            }
49  	        }
50  	        
51  	        return graph;
52  	    }

54  	    /**
55  	     * Helper method to get an existing vertex or create a new one if needed
56  	     * @param Map<String, Vertex> verticesMap
57  	     * @param Graph graph
58  	     * @param String id
59  	     * @return Vertex
60  	     */
61  	    private static Vertex getOrCreateVertex(Map<String, Vertex> verticesMap, Graph graph, String id) {
62  	        Vertex vertex = verticesMap.get(id);
63  	        if (vertex == null) {
64  	            vertex = new Vertex(id);
65  	            verticesMap.put(id, vertex);
66  	            graph.addVertex(vertex);
67  	        }
68  	        return vertex;
69  	    }

71  	    /**
72  	     * Reconstructs the shortest path from source to destination using the predecessor map
73  	     * 
74  	     * @param pred The predecessor map from Dijkstra's algorithm
75  	     * @param source The source vertex
76  	     * @param destination The destination vertex
77  	     * @return A list of vertices representing the path from source to destination
78  	     */
79  	    private static List<Vertex> reconstructPath(TreeMap<Vertex, Vertex> pred, Vertex source, Vertex destination) {
80  	        List<Vertex> path = new ArrayList<>();
81  	        
82  	        // Start from the destination and work backwards
83  	        for (Vertex current = destination; current != null; current = pred.get(current)) {
84  	            path.add(current);
85  	            if (current.equals(source)) {
86  	                break; // We've reached the source
87  	            }
88  	        }
89  	        
90  	        // Reverse the path to get source to destination
91  	        Collections.reverse(path);
92  	        return path;
93  	    }
