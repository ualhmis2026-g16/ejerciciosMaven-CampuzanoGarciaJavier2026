3   import java.io.*;
4   import java.util.*;
5   
6   public class MainEj5EDALandLarge {
7   	/**
8   	 * metodo main que se encarga de realizar la ejecucion de DijkstraEj5
9   	 * con el fichero graphEDAlandLarge
10  	 * @param Strin[] args
11  	 */
12      public static void main(String[] args) {
13          Graph graph = new Graph();
14          Map<String, Vertex> vertexMap = new HashMap<>();
15          Map<Vertex, double[]> positions = new HashMap<>(); // Mapa de coordenadas para la heurística
16  
17          // Directorio de entrada
18          String directorioEntrada = System.getProperty("user.dir") + File.separator +
19                  "src" + File.separator +
20                  "main" + File.separator +
21                  "java" + File.separator +
22                  "ual" + File.separator +
23                  "eda2" + File.separator +
24                  "practica02" + File.separator;
25  
26          String filename = directorioEntrada + "graphEDAlandLarge.txt";
27          String positionsFile = directorioEntrada + "graphEDAlandLargePositions.txt"; // Archivo de coordenadas
28  
29          // Leer archivo de grafo y construirlo
30          try (BufferedReader br = new BufferedReader(new FileReader(filename))) {
31              String line;
32              while ((line = br.readLine()) != null) {
33                  if (line.startsWith("#") || line.trim().isEmpty()) continue;
34  
35                  String[] parts = line.trim().split("\\s+");
36                  if (parts.length < 3) continue;
37  
38                  String fromId = parts[0];
39                  String toId = parts[1];
40                  double weight = Double.parseDouble(parts[2]);
41  
42                  Vertex from = vertexMap.computeIfAbsent(fromId, Vertex::new);
43                  Vertex to = vertexMap.computeIfAbsent(toId, Vertex::new);
44  
45                  graph.addEdge(from, to, weight);
46              }
47          } catch (IOException e) {
48              System.err.println("Error leyendo el archivo del grafo: " + e.getMessage());
49              return;
50          }
51  
52          // Leer archivo de posiciones para la heurística
53          try (BufferedReader br = new BufferedReader(new FileReader(positionsFile))) {
54              String line;
55              while ((line = br.readLine()) != null) {
56                  if (line.startsWith("#") || line.trim().isEmpty()) continue;
57  
58                  String[] parts = line.trim().split("\\s+");
59                  if (parts.length < 3) continue;
60  
61                  String vertexId = parts[0];
62                  double x = Double.parseDouble(parts[1]);
63                  double y = Double.parseDouble(parts[2]);
64  
65                  Vertex vertex = vertexMap.computeIfAbsent(vertexId, Vertex::new);
66                  positions.put(vertex, new double[]{x, y});
67              }
68          } catch (IOException e) {
69              System.err.println("Error leyendo el archivo de posiciones: " + e.getMessage());
70              return;
71          }
72  
73          // Solicitar al usuario un vértice de origen y destino
74          Scanner scanner = new Scanner(System.in);
75          System.out.print("Introduce el ID del vértice de origen: ");
76          String origenId = scanner.nextLine().trim();
77  
78          System.out.print("Introduce el ID del vértice de destino: ");
79          String destinoId = scanner.nextLine().trim();
80  
81          // Verificar si el vértice de origen y destino existen
82          Vertex source = vertexMap.get(origenId);
83          Vertex destination = vertexMap.get(destinoId);
84  
85          if (source == null) {
86              System.err.println("El vértice de origen no existe en el grafo.");
87              return;
88          }
89          if (destination == null) {
90              System.err.println("El vértice de destino no existe en el grafo.");
91              return;
92          }
93  
94          // Ejecutar el algoritmo A* (DijkstraEj5)
95          System.out.println("\nCalculando camino mínimo desde " + source.getId() + " hasta " + destination.getId() + "...\n");
96          TreeMap<String, Object> results = DijkstraEj5.findPath(graph, source, destination, positions);
97  
98          // Recuperar resultados
99          @SuppressWarnings("unchecked")
100         List<Vertex> path = (List<Vertex>) results.get("path");
101         double distance = (double) results.get("distance");
102         boolean pathFound = (boolean) results.get("pathFound");
103         long tiempoEjecucion = (long) results.get("tiempo_ejecucion_ns");
104         int nodosEvaluados = (int) results.get("nodos_evaluados");
105         int nodosGenerados = (int) results.get("nodos_generados");
106 
107         // Mostrar resultados
108         if (!pathFound) {
109             System.out.println("No existe camino desde " + source.getId() + " hasta " + destination.getId());
110         } else {
111             System.out.println("Camino mínimo encontrado:");
112             System.out.printf("Distancia total: %.2f\n", distance);
113             System.out.print("Ruta: ");
114             for (int i = 0; i < path.size(); i++) {
115                 System.out.print(path.get(i).getId());
116                 if (i < path.size() - 1) {
117                     System.out.print(" -> ");
118                 }
119             }
120             System.out.println();
121         }
122 
123         // Mostrar estadísticas
124         System.out.println("\nEstadísticas de ejecución:");
125         System.out.printf("Tiempo de ejecución: %d ns\n", tiempoEjecucion);
126         System.out.printf("Nodos evaluados: %d\n", nodosEvaluados);
127         System.out.printf("Nodos generados: %d\n", nodosGenerados);
128 
129         scanner.close();
130     }
131 }
