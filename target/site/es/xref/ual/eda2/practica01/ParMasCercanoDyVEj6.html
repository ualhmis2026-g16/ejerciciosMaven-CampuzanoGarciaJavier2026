14  	 * Comparador de puntos según la coordenada X
15  	 */
16  	static Comparator<Poin> compareX = new Comparator<Poin>() {
17  		@Override
18  		public int compare(Poin p1, Poin p2) {
19  			return Double.compare(p1.x, p2.x);
20  		}
21  	};
22  	
23  	/*
24  	 * Comparador de puntos según la coordenada Y
25  	 */
26  	static Comparator<Poin> compareY = new Comparator<Poin>() {
27  		@Override
28  		public int compare(Poin p1, Poin p2) {
29  			return Double.compare(p1.y, p2.y);
30  		}
31  	};
32  
33  	/**
34       * Calcula la distancia euclídea entre un par de puntos dado
35       * 
36       * @param p1 Primer punto
37       * @param p2 Segundo punto
38       * @return Distancia euclídea entre los dos puntos
39       */
40  	public static double distancia(Poin p1, Poin p2) {
41  		return Math.sqrt((p1.x - p2.x) * (p1.x - p2.x) +
42  						(p1.y - p2.y) * (p1.y - p2.y));
43  	}
44  	
45  	/**
46       * Método para encontrar la distancia entre los puntos más cercanos 
47       * de una lista haciendo uso del algoritmo de Divide y Vencerás, con la última mejora:
48       * Se crean 2 copias del array de puntos: Px, ordenado por la coordenada X, y Py, ordenado por la coordenada Y.
49       * Tras esto, cada array se divide en 2: Plx, Prx, Ply, Pry. Estos 4 arrays, básicamente dividen a los arrays anteriores
50       * según su posición con respecto a la línea vertical que divide a todos los puntos por la mitad.
51       * De esta manera, se van dividiendo en 4 los arrays de puntos que entran, para luego seguir el mismo procedimiento con la franja.
52       * 
53       * @param listaPuntos Los puntos a comparar
54       * @return Distancia mínima entre los puntos más cercanos de la lista
55       */
56  	public static double parMasCercanoDivideandConquer(ArrayList<Poin> listaPuntos) {
57  		ArrayList<Poin> puntosejey = new ArrayList<>(listaPuntos);
58  		Collections.sort(puntosejey, compareY);
59  	    Collections.sort(listaPuntos, compareX);
60  	    return parMasCercanoDivideandConquer(listaPuntos, puntosejey, 0, listaPuntos.size() - 1);
61  	}
62  
63  	/**
64       * Parte privada del recursivo.
65       * 
66       * @param Px       Lista de puntos ordenados por la coordenada X.
67       * @param Py       Lista de puntos ordenados por la coordenada Y.
68       * @param comienzo Índice inicial del subarray a considerar.
69       * @param fin      Índice final del subarray a considerar.
70       * @return Distancia mínima entre dos puntos dentro del subarray.
71       */
72  	private static double parMasCercanoDivideandConquer(ArrayList<Poin> Px, ArrayList<Poin> Py, int comienzo, int fin) {
73  		
74  		//Caso base, fuerza bruta cuando el array de puntos es de tamaño 3 o inferior.
75  		
76  	    if (fin - comienzo <= 3) {
77  	    	double minDist = Double.MAX_VALUE;
78  			double aux;
79  			for(int i = comienzo; i<=fin; i++) {
80  				for(int j=i+1; j<=fin; j++) {
81  					aux = distancia(Px.get(i), Px.get(j));
82  					if(aux<minDist) {
83  						minDist = aux;
84  					}
85  				}
86  			}
87  			return minDist;
88  	    } else {
89  	    	
90  	    	//Dividir
91  	    	
92  	        int mitad = comienzo + (fin - comienzo) / 2;
93  	        double lineaX = Px.get(mitad).x;
94  	        
95  	        //Combinar
96  	        
97  	        ArrayList<Poin> plx = new ArrayList<>();
98  	        ArrayList<Poin> prx = new ArrayList<>();
99  	        ArrayList<Poin> ply = new ArrayList<>();
100 	        ArrayList<Poin> pry = new ArrayList<>();
101 	        
102 	        for (int i = comienzo; i <= fin; i++) {
103 	            if (i <= mitad) {
104 	                plx.add(Px.get(i));
105 	            } else {
106 	                prx.add(Px.get(i));
107 	            }
108 	        }
109 	        
110 	        for (Poin p : Py) {
111 	            if (p.x <= lineaX) {
112 	                ply.add(p);
113 	            } else {
114 	                pry.add(p);
115 	            }
116 	        }
117 	        
118 	        double dl = parMasCercanoDivideandConquer(plx, ply, 0, plx.size() - 1);
119 	        double dr = parMasCercanoDivideandConquer(prx, pry, 0, prx.size() - 1);
120 
121 	        double delta = Math.min(dl, dr);
122 	        
123 	        ArrayList<Poin> franja = new ArrayList<>();
124 	        for (Poin p : Py) {
125 	            if (Math.abs(p.x - lineaX) < delta) {
126 	                franja.add(p);
127 	            }
128 	        }
129 	        
130 	        double aux;
131 	        for (int i = 0; i < franja.size(); i++) {
132 	            for (int j = i + 1; j < franja.size() && j<=i+7; j++) {
133 	                aux = distancia(franja.get(i), franja.get(j));
134 	                if (aux < delta) {
135 	                    delta = aux;
136 	                }
137 	            }
138 	        }
139 	        
140 	        return delta;
141 	    }
142 	}
